💬 Doğal Dilden SQL'e Geçiş

Snowflake Intelligence ve Looker'ın Conversational Analytics özellikleri, teknik olmayan kullanıcıların karmaşık veritabanlarını doğal dil ile sorgulamasına olanak tanıyor.

-- Geleneksel SQL sorgusu
SELECT 
    customer_id,
    SUM(order_amount) as total_spent,
    COUNT(*) as order_count
FROM orders 
WHERE order_date >= '2026-01-01'
GROUP BY customer_id
HAVING total_spent > 1000;

-- Doğal dil sorgusu: "2026'da 1000$'dan fazla harcama yapan müşterileri göster"

# Finansal veri analizi için Python örneği
import pandas as pd
import numpy as np
import yfinance as yf
from sklearn.ensemble import RandomForestRegressor
from sklearn.model_selection import train_test_split

# Hisse senedi verilerini çekme
ticker = "AAPL"
data = yf.download(ticker, start="2020-01-01", end="2026-01-01")

# Teknik göstergeler ekleme
data['SMA_20'] = data['Close'].rolling(window=20).mean()
data['SMA_50'] = data['Close'].rolling(window=50).mean()
data['RSI'] = calculate_rsi(data['Close'])

# Özellik mühendisliği
data['Price_Change'] = data['Close'].pct_change()
data['Volume_Change'] = data['Volume'].pct_change()

# Makine öğrenmesi modeli için veri hazırlama
features = ['SMA_20', 'SMA_50', 'RSI', 'Volume_Change']
target = 'Price_Change'

# Eksik değerleri temizle
data_clean = data.dropna()

# Model eğitme
X_train, X_test, y_train, y_test = train_test_split(
    data_clean[features], 
    data_clean[target], 
    test_size=0.2, 
    random_state=42
)

model = RandomForestRegressor(n_estimators=100, random_state=42)
model.fit(X_train, y_train)

print(f"Model Doğruluğu: {model.score(X_test, y_test):.2f}")

Modern SQL Sorgu Örneği

-- Finansal analiz için gelişmiş SQL sorgusu
WITH customer_metrics AS (
    SELECT 
        customer_id,
        SUM(transaction_amount) as total_spent,
        COUNT(*) as transaction_count,
        AVG(transaction_amount) as avg_transaction,
        MAX(transaction_date) as last_transaction,
        -- RFM analizi
        CASE 
            WHEN SUM(transaction_amount) > 1000 THEN 'VIP'
            WHEN SUM(transaction_amount) > 500 THEN 'Premium'
            ELSE 'Standard'
        END as customer_segment
    FROM transactions 
    WHERE transaction_date >= CURRENT_DATE - INTERVAL '90 days'
    GROUP BY customer_id
),
segment_analysis AS (
    SELECT 
        customer_segment,
        COUNT(*) as customer_count,
        AVG(total_spent) as avg_spending,
        SUM(total_spent) as total_revenue
    FROM customer_metrics
    GROUP BY customer_segment
    ORDER BY total_revenue DESC
)
SELECT * FROM segment_analysis;
